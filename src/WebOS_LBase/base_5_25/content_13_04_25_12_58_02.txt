void sor(Node **head)

 {

     Node *p=*head;

     Node *q,*s;

     if(!p||!p->next)//head=NULL，或者只有一个元素返回

         return ;

     q=p->next;

     s=q->next;

     while(q)

     {

         q->next=p;

         p=q;

         q=s;

         if(!s);

         else

         s=s->next;

     }

     (*head)->next=NULL;

     *head=p;

 }

int getLast(int iLen)

 {

     int *data=new int[iLen];

     for(int i=0;i<iLen;i++)

         cin>>data[i];

     int cnt=iLen,index=0,flag=1,last;

     while(cnt>1)

     {

         ++index;

         if(data[index%iLen]!=-1)

         ++flag;

         if(flag==3)

         {

             data[index%iLen]=-1;

             flag=0;

             --cnt;//将数组无数减1

         }

     }

     for(int j=0;j<iLen;j++)//这种方法效率不高，要再一次循环，求坐标

         if(data[j]!=-1)

             last=j;

     return last;

 }
